nu2017 <- winsloss2017 |>
  filter (Team == "Nebraska (Big Ten)")

winsloss <- bind_rows(winsloss2017, winsloss2018, winsloss2019, winsloss2020, winsloss2021, winsloss2022, winsloss2023) |>
filter(Rank <= 25) 

nuwinsloss <- winsloss |> filter(Team == "Nebraska (Big Ten)")

kentuckywinsloss <- winsloss |> filter(Team == "Kentucky (SEC)")

texaswinsloss <- winsloss |> filter(Team == "Texas (Big 12)")

ggplot() +
  geom_line(data=winsloss, aes(x=Season, y=`Pct.`, group=Team), color="light grey") +
  geom_line(data=nuwinsloss, aes(x=Season, y=`Pct.`, group=Team), color="red") +
  geom_point(data=nuwinsloss, aes(x=Season, y=`Pct.`, group=Team), color="red") +

geom_line(data=texaswinsloss, aes(x=Season, y=`Pct.`, group=Team), color="#BF5700") +
labs(x="Season", y="Win Pct.", title="Nebraska's on top, but hasn't always been", subtitle="Nebraska, Kentucky and Texas are some of the several national championship winners in the last few years.
